Env setup (PowerShell):

Ensure .gitignore contains .env (create if missing). Never commit .env.
If .env exists, load it into the current process env (do NOT echo secret values):
if (!(Test-Path .gitignore)) { ".env" | Out-File -Encoding utf8 .gitignore }
elseif (-not (Select-String -Quiet -Path .gitignore -Pattern '^\.env$')) { "`n.env" | Add-Content -Encoding utf8 .gitignore }

if (Test-Path .env) {
  Get-Content .env | ForEach-Object {
    $l = $_.Trim()
    if (-not $l -or $l.StartsWith('#') -or ($l -notmatch '=')) { return }
    $k,$v = $l -split '=',2
    $k = $k.Trim()
    $v = $v.Trim().Trim('"')
    Set-Item -Path ("Env:" + $k) -Value $v
  }
}
